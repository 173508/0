    Filename Fix for converting spaces to underscore in Bash¶

    Reference https://stackoverflow.com/a/18213120

    for f in *\ *; do mv "$f" "${f// /_}"; done

    This magic one liner will replace all the spaces in the file names in the current directory with _ underscores.

    The ${f// /_} uses the special Parameter Expansion of Bash to perform the string replacement.

    The relevant syntax is ${parameter/pattern/string}.

    See: https://www.gnu.org/software/bash/manual/html_node/Shell-Parameter-Expansion.html or http://wiki.bash-hackers.org/syntax/pe .

    Here is also a Bash Function just in case:

    function unspacefile() {
for f in *\ *; do mv "$f" "${f// /_}"; done
}

    Input Scripts for Bash¶

    ####
## Environment Variables
#
export ANS=''
####
## Input function for [No/Yes/Quit] with default option
## (if not supplied being "No").
##
## This function returns the result in environment variable
## $ANS
##
## There can be 3 types of return values
## 'y' = For any positive inputs
## 'n' = For any negative inputs
## 'q' = To exit the Process
##
## usage:
##    inputnyq "<Message>" ["<Default-value>"]
##
#
inputnyq () {
    export ANS=''
    local MSG=$1
    local DEFAULT=$2
    local ans=''
    if [ -z "$MSG" ]; then
    MSG="Enter Choice"
    fi
    if [ -z "$DEFAULT" ]; then
    DEFAULT="n"
    fi
    echo -n "$MSG (y/n/q)[$DEFAULT]: "
    read -r -e ans
    case $ans in
    Q|q|Quit|exit|x|Exit|E) ans='q';;
    Y|y|yes|Yes|ok|Ok|OK|YES) ans='y';;
    n|N|No|no|NO|na) ans='n';;
    *) ans="$DEFAULT";;
    esac
    export ANS="$ans"
}
####
## Input function similar to 'inputnyq'
## for 'q' return it quits the execution.
##
## This function returns the result in environment variable
## $ANS
##
## usage:
##    inpute "<Message>" ["<Default-value>"]
#
inpute () {
    inputnyq "$@"
    if [ "$ANS" == "q" ]; then
    echo
    echo " Qutting !!"
    echo
    exit
    fi
}
####
## Input function with Default value used for user input values.
## Here default value is mandatory.
##
## This function returns the result in environment variables.
## $ANS
##
## usage:
##    inputd "<Message>" "<Default-value>"
#
inputd () {
    export ANS=''
    local MSG=$1
    local DEFAULT=$2
    if [ -z "$MSG" ]; then
    MSG="Enter Value"
    fi
    echo -n "$MSG [$DEFAULT]: "
    read -r -e ans
    if [ -z "$ans" ]; then
    ans="$DEFAULT"
    fi
    export ANS="$ans"
}
## Read response
## Returns the value on screen so needs to captured as a variable:
##
## printf "    Do you want to Fix Fail to Lock (y/n)[n]? "
## ...
## ans=$(read_response)
##
function read_response()
{
    local ans=""
    read -r ans
    case "$ans" in
    "y"|"Y"|"yes"|"Yes") echo "yes";;
    *) echo "no";;
    esac
}

    Backup Directory to a Archive with Date and Time stamp in Bash¶

    # Backup function
backup_directory() {
  # Ensure a directory is passed as an argument
  if [ -z "$1" ]; then
    echo "Usage: backup_directory <source_directory>"
    return 1
  fi

  # Get the source directory
  local src_dir=$(eval echo "$1")

  # Check if the source directory exists
  if [ ! -d "$src_dir" ]; then
    echo "Directory '$src_dir' does not exist."
    return 1
  fi

  # Navigate to the source directory
  local base_dir=$(basename "$src_dir")
  # Adjusted time format (YYYY-MM-DDTHH.MM.SS)
  local timestamp=$(date +"%Y-%m-%dT%H.%M.%S")

  # Create the tarball backup filename
  local backup_filename="${base_dir}-${timestamp}.tar.gz"

  echo "Backing up $1 into $backup_filename..."
  # Perform the backup inside the source directory environment
  tar -czf "$backup_filename" -C "$src_dir"  .

  # Check if tar command was successful
  if [ $? -eq 0 ]; then
    echo "Backup successful: $backup_filename"
  else
    echo "Backup failed."
    return 1
  fi
}

    How to use this:

    echo
src="~/Desktop"

##############
# Backup Project folders
backup_directory "$src/proj1"
backup_directory "$src/proj2"

echo " Done !!"
echo

    The above code would produce two files, for Example:

    1. proj1-2024-10-12T10.12.14.tar.gz
    2. proj2-2024-10-12T10.12.14.tar.gz

    This the date and timestamp marked file for easily transporting and backing up.

    Synchronizing Directories using rsync command in Bash¶

    Here are the functions:

    Fast Syncing Function using rsync with its Dry run counter part¶

    targetdrive="/mnt/hdd-mount"

function dryrunfastsync () {
    echo "---- Start Dry Run : $1 to ${targetdrive}/$1/ ----"
    rsync -avhPi --delete --force --dry-run --info=progress2 "./$1/" "${targetdrive}/$1/"
    echo "----  End  Dry Run : $1 to ${targetdrive}/$1/ ----"
    echo
}

function fastsync() {
    echo ">>>> Start Fast Sync : $1 to ${targetdrive}/$1/ <<<<"
    #read -r -p "Press Enter to Begin...."
    rsync -avhPi --delete --force --info=progress2 "./$1/" "${targetdrive}/$1/"
    echo ">>>>  End  Fast Sync : $1 to ${targetdrive}/$1/ <<<<"
    echo
}

    This function would need targetdrive for location to where the synchronization needs to take place.

    It is the parent directory of the directory name specified in $1 parameter.

    This process assumes that source and target directories are organized as follows:

    • Source Directory ./$1 - This is where the command is being executed.
    • Destination Directory $targetdrive/$1

    Here is an Example of operation:

    export targetdrive="/media/My-HDD"

echo
dryrunfastsync Projects-Archive

read -p "Ready press enter to begin actual Sync..."
fastsync Projects-Archive

    In this case the current directory holds Projects-Archive directory and /media/My-HDD/Projects-Archive is the target location.

    Fast Syncing with different directory Names using rsync¶

    Sometimes source and target directory names are not the same.

    Hence another alternative method is used:

    targetdrive="/media/hdd-mount"

function dryrunfastsync2 () {
    echo "---- Start Dry Run : $1 to ${targetdrive}/$2/ ----"
    rsync -avhPi --delete --force --dry-run --info=progress2 "./$1/" "${targetdrive}/$2/"
    echo "----  End  Dry Run : $1 to ${targetdrive}/$2/ ----"
    echo
}

function fastsync2() {
    echo ">>>> Start Fast Sync : $1 to ${targetdrive}/$2/ <<<<"
    #read -r -p "Press Enter to Begin...."
    rsync -avhPi --delete --force --info=progress2 "./$1/" "${targetdrive}/$2/"
    echo ">>>>  End  Fast Sync : $1 to ${targetdrive}/$2/ <<<<"
    echo
}

    In this case it still uses the ${targetdrive} for the parent of the synchronization target directory. But the source and target names are are now specified in $1 and $2 respectively.

    Here is an Example:

    export targetdrive="/media/My-HDD"

echo
dryrunfastsync2 Projects-Archive work/Office-Projects

read -p "Ready press enter to begin actual Sync..."
fastsync2 Projects-Archive work/Office-Projects

    Here is how the source and target now looks like:

    • Source ./Projects-Archive
    • Target /media/My-HDD/work/Office-Projects

    This way we can specify different names.

    Checksum Enabled Syncing with different options using rsync¶

    Sometime we need extra reliability of Checksum enabled synchronization. This helps to minimize any damage or loss during transmission and storage of data.

    The only demerit of this technique is that it takes much more time to execute than Fast Sync.

    targetdrive="/mnt/hdd-mount"

function dryrunchecksum () {
    echo "---- Start Dry Run with Checksum : $1 to ${targetdrive}/$1/ ----"
    rsync -avchPi --cc=xxh128 --delete --force --info=progress2 --dry-run "./$1/" "${targetdrive}/$1/"
    echo "----  End  Dry Run with Checksum : $1 to ${targetdrive}/$1/ ----"
    echo
}

function checksumsync () {
    echo "==== Start Checksum Sync : $1 to ${targetdrive}/$1/ ===="
    #read -r -p "Press Enter to Begin...."
    rsync -avchPi --cc=xxh128 --delete --force --info=progress2 "./$1/" "${targetdrive}/$1/"
    echo "====  End  Checksum Sync : $1 to ${targetdrive}/$1/ ===="
    echo
}

function dryrunchecksum2 () {
    echo "---- Start Dry Run with Checksum : $1 to ${targetdrive}/$2/ ----"
    rsync -avchPi --cc=xxh128 --delete --force --info=progress2 --dry-run "./$1/" "${targetdrive}/$2/"
    echo "----  End  Dry Run with Checksum : $1 to ${targetdrive}/$2/ ----"
    echo
}

function checksumsync2 () {
    echo "==== Start Checksum Sync : $1 to ${targetdrive}/$2/ ===="
    #read -r -p "Press Enter to Begin...."
    rsync -avchPi --cc=xxh128 --delete --force --info=progress2 "./$1/" "${targetdrive}/$2/"
    echo "====  End  Checksum Sync : $1 to ${targetdrive}/$2/ ===="
    echo
}

    These are both the types of synchronization and uses xxh128 checksum.

    Here is how to use them:

    export targetdrive="/media/My-HDD"

echo
dryrunchecksum Projects-Archive

read -p "Ready press enter to begin actual Sync..."
checksumsync Projects-Archive

    Again here are the directories:

    • Source ./Projects-Archive
    • Target /media/My-HDD/Projects-Archive

    Alternatively:

    export targetdrive="/media/My-HDD"

echo
dryrunchecksum2 Projects-Archive work/Office-Projects

read -p "Ready press enter to begin actual Sync..."
checksumsync2 Projects-Archive work/Office-Projects

    Here the directories are :

    • Source ./Projects-Archive
    • Target /media/My-HDD/work/Office-Projects
