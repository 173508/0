    ESP-NOW Communication - ESP8266 (Arduino Core)¶

    ESP-NOW is a connectionless communication protocol by Espressif, ideal for low-latency, low-power, short-range peer-to-peer messaging — particularly useful for IoT.

    This guide focuses on ESP8266 using the official Arduino Core, and includes setup, communication patterns (1-way, 2-way, many-to-one, etc.), limitations, and advanced tips.

    Getting Started¶

    Requirements¶

    • ESP8266 board (NodeMCU, Wemos D1 Mini, etc.)
    • ESP8266 Arduino Core
    • Arduino IDE or PlatformIO

    Installing ESP-NOW Support¶

    ESP-NOW is built into the ESP8266 Arduino Core — no extra libraries required.

    #include <ESP8266WiFi.h>
#include <espnow.h>

    Make sure your sketch disables Wi-Fi connection setup. ESP-NOW requires the ESP8266 to be in station mode or softAP + station mode.

    WiFi.mode(WIFI_STA); // Or WIFI_AP_STA for more complex setups

    Communication Modes¶

    ESP-NOW allows multiple flexible topologies:

    One-Way (Unidirectional)¶

    A → B (No ACK required)

    Sender (Master):

    #include <ESP8266WiFi.h>
#include <espnow.h>

uint8_t slaveMac[] = {0xDE, 0xAD, 0xBE, 0xEF, 0xFE, 0xED};

void setup() {
  Serial.begin(115200);
  WiFi.mode(WIFI_STA);
  if (esp_now_init() != 0) {
    Serial.println("ESP-NOW init failed");
    return;
  }
  esp_now_set_self_role(ESP_NOW_ROLE_CONTROLLER);
  esp_now_add_peer(slaveMac, ESP_NOW_ROLE_SLAVE, 1, NULL, 0);
  uint8_t msg[] = {42};
  esp_now_send(slaveMac, msg, sizeof(msg));
}

void loop() {}

    Receiver (Slave):

    #include <ESP8266WiFi.h>
#include <espnow.h>

void onReceive(uint8_t *mac, uint8_t *data, uint8_t len) {
  Serial.print("Received: ");
  Serial.println(*data);
}

void setup() {
  Serial.begin(115200);
  WiFi.mode(WIFI_STA);
  if (esp_now_init() != 0) {
    Serial.println("ESP-NOW init failed");
    return;
  }
  esp_now_set_self_role(ESP_NOW_ROLE_SLAVE);
  esp_now_register_recv_cb(onReceive);
}

void loop() {}

    Two-Way (Bidirectional)¶

    A ⇄ B — Both devices send and receive.

    Device A and B (same logic on both ends):

    #include <ESP8266WiFi.h>
#include <espnow.h>

uint8_t peerMac[] = {0xDE, 0xAD, 0xBE, 0xEF, 0xFE, 0xED}; // Replace with other device's MAC

void onReceive(uint8_t *mac, uint8_t *data, uint8_t len) {
  Serial.print("Received: ");
  Serial.println(*data);
}

void onSend(uint8_t *mac, uint8_t status) {
  Serial.println(status == 0 ? "Sent OK" : "Send Failed");
}

void setup() {
  Serial.begin(115200);
  WiFi.mode(WIFI_STA);
  if (esp_now_init() != 0) {
    Serial.println("ESP-NOW init failed");
    return;
  }
  esp_now_set_self_role(ESP_NOW_ROLE_COMBO);
  esp_now_add_peer(peerMac, ESP_NOW_ROLE_COMBO, 1, NULL, 0);
  esp_now_register_recv_cb(onReceive);
  esp_now_register_send_cb(onSend);

  uint8_t data[] = {99};
  esp_now_send(peerMac, data, sizeof(data));
}

void loop() {}

    One-to-Many¶

    A → B, C, D...

    Sender adds multiple peers and sends individually to each.

    Broadcast Communication¶

    While ESP8266 does not support native broadcast with encryption, you can simulate a broadcast by sending to the universal MAC address FF:FF:FF:FF:FF:FF. This method does not require the receiver to be registered as a peer, and is useful for one-to-many messages without acknowledgments.

    Broadcast Sender Example:

    #include <ESP8266WiFi.h>
#include <espnow.h>

uint8_t broadcastAddress[] = {0xFF, 0xFF, 0xFF, 0xFF, 0xFF, 0xFF};

void setup() {
  Serial.begin(115200);
  WiFi.mode(WIFI_STA);
  if (esp_now_init() != 0) {
    Serial.println("ESP-NOW init failed");
    return;
  }
  esp_now_set_self_role(ESP_NOW_ROLE_CONTROLLER);

  uint8_t data[] = {1};
  esp_now_send(broadcastAddress, data, sizeof(data));
}

void loop() {}

    Broadcast Receiver (any listening ESP8266):

    #include <ESP8266WiFi.h>
#include <espnow.h>

void onReceive(uint8_t *mac, uint8_t *data, uint8_t len) {
  Serial.print("Received broadcast: ");
  Serial.println(*data);
}

void setup() {
  Serial.begin(115200);
  WiFi.mode(WIFI_STA);
  if (esp_now_init() != 0) {
    Serial.println("ESP-NOW init failed");
    return;
  }
  esp_now_set_self_role(ESP_NOW_ROLE_SLAVE);
  esp_now_register_recv_cb(onReceive);
}

void loop() {}

    Note: Since this uses broadcast, you cannot receive acknowledgments, and it may be unreliable in noisy environments.

    Many-to-One¶

    A, B, C → D

    D registers each sender and listens.

    Many-to-Many¶

    Each node registers every other node and can send/receive messages with them.

    Arbitrary Pattern¶

    Peers can be added or removed dynamically at runtime.

    esp_now_del_peer(oldMac);
esp_now_add_peer(newMac, ESP_NOW_ROLE_COMBO, 1, NULL, 0);

    Limitations (ESP8266 vs ESP32)¶

    Feature ESP8266 ESP32
    Max Peers ~20 250+
    Encryption No (not supported) Yes (supported)
    Broadcast Limited (manual) Native support
    Power Efficiency Medium Better for battery
    Range ~50-70 meters typical ~100+ meters possible

    Note: ESP8266 does not support encrypted ESP-NOW, unlike ESP32.

    Tips & Tricks¶

    • Use WIFI_AP_STA if you need simultaneous AP + ESP-NOW behavior.
    • Debounce or queue your messages — no retransmissions are built-in.
    • Use WiFi.macAddress() to configure peers.
    • Use structs for payloads, but ensure both sides use the same structure.
    • Avoid WiFi.begin() unless you're switching to full Wi-Fi.

    Conclusion¶

    ESP-NOW offers lightweight, connectionless messaging ideal for sensor networks, remote control, and low-power communication. While ESP8266 has some limitations compared to ESP32, it's still powerful for many use-cases.

    ESP8266 as ESP-NOW Master (Send Only) Notes¶

    This is a Send only configuration for ESP8266. It needs another board to be in the role as a Receiver to get the data sent.

    Note: We are not selecting any CHANNEL here in the code its assumed to be 1 since there is no way to change this in Master.

    Includes and Globals¶

    // Includes
#include <ESP8266WiFi.h>
#include <espnow.h>

// REPLACE WITH YOUR RECEIVER MAC Address
uint8_t broadcastAddress[] = {0xFF, 0xFF, 0xFF, 0xFF, 0xFF, 0xFF};
// uint8_t broadcastAddress[] = {0x01, 0x02, 0x03, 0x04, 0x05, 0x06};

String myData = "Test";

    Basic Setup¶

    // Setup

// For Maximum Message length of ESP-NOW + 1
myData.reserve(251);

// Set device as a Wi-Fi Station
WiFi.mode(WIFI_STA); // Selects Channel 0 by default

// Initialize ESP-NOW
if (esp_now_init() != 0) {
  Serial.println("Error initializing ESP-NOW");
  return;
} else {
  Serial.println("Initialized ESP-NOW");
}

    Configure ESP-NOW¶

    // Setup continued

  // Once ESPNow is successfully Init, we will register for Send CB to
  // get the status of Transmitted packet
  esp_now_set_self_role(ESP_NOW_ROLE_CONTROLLER);
  esp_now_register_send_cb(OnDataSent);
    Here OnDataSent is a Call Back function.

    Auto populate the MAC ID of Slave¶

    // Find a Single slave to talk to directly
bool ScanForSlave() {
  int8_t scanResults = WiFi.scanNetworks();

  // Find the Networks
  for (int i = 0; i < scanResults; ++i) {
    String SSID = WiFi.SSID(i);
    String BSSIDstr = WiFi.BSSIDstr(i);

    // Identifier for the Slave
    if (SSID.indexOf("RX_") == 0) {
      Serial.print("Found Slave -> ");
      Serial.println(BSSIDstr.c_str());
      int mac[6];
      if ( 6 == sscanf(BSSIDstr.c_str(), "%x:%x:%x:%x:%x:%x",
        &mac[0], &mac[1], &mac[2], &mac[3], &mac[4], &mac[5] ) ) {
        for (int ii = 0; ii < 6; ++ii ) {
          broadcastAddress[ii] = (uint8_t) mac[ii];
        }
      }

      return true;
    }
  }
  return false;
}
